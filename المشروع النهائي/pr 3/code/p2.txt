    <script>
        document.getElementById('accountType').addEventListener('change', function() {
            var accountType = this.value;
            var ownerFields = document.getElementById('ownerFields');
            var tenantFields = document.getElementById('tenantFields');

            if (accountType === 'owner') {
                ownerFields.style.display = 'block';
                tenantFields.style.display = 'none';
            } else if (accountType === 'tenant') {
                tenantFields.style.display = 'block';
                ownerFields.style.display = 'none';
            }
        });

        document.addEventListener('DOMContentLoaded', function() {
            var accountType = document.getElementById('accountType').value;
            var ownerFields = document.getElementById('ownerFields');
            var tenantFields = document.getElementById('tenantFields');

            if (accountType === 'owner') {
                ownerFields.style.display = 'block';
                tenantFields.style.display = 'none';
            } else if (accountType === 'tenant') {
                tenantFields.style.display = 'block';
                ownerFields.style.display = 'none';
            }
        });

        function validateForm() {
            var password = document.getElementById('password').value;
            var confirmPassword = document.getElementById('confirmPassword').value;
            if (password !== confirmPassword) {
                alert("كلمة المرور وتأكيد كلمة المرور غير متطابقين.");
                return false;
            }
            return true;
        }
    </script>
